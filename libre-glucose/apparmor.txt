#include <tunables/global>

# Profile name MUST match the app slug. Supervisor loads this file
# automatically when `apparmor: true` is set in config.yaml.
profile libre-glucose flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # ---------------------------------------------------------------------
  # s6-overlay / bashio scaffolding (HA add-on base image)
  # ---------------------------------------------------------------------
  # s6-overlay v3 (in the HA *-base-debian:bookworm images) ships /init
  # and most of /package, /command as `#!/bin/sh` shebang scripts — not
  # ELF binaries like s6-overlay v2 did. The Linux kernel must
  # open(O_RDONLY) a shebang script to read its `#!`-line during
  # execve(), so AppArmor `ix` (execute + inherit, no read) causes
  # EACCES and the well-known boot failure:
  #   /bin/sh: 0: cannot open /init: Permission denied
  # See: AppArmor docs ("Read access is required for shell scripts and
  # other interpreted content"), HA Supervisor #1748, s6-overlay v2→v3
  # base-image migration (HA dev blog, May 2022).
  # ELF paths (/bin, /usr/bin, /usr/sbin, /usr/local/bin) stay `ix` —
  # ELF exec does not require a userspace read; granting `r` would only
  # widen the read surface for no functional gain.
  /init rix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /usr/local/bin/** ix,
  /run/{s6,s6-rc*,service}/** rix,
  /package/** rix,
  /command/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /etc/s6-overlay/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio helpers — sourced (`. /usr/lib/bashio/bashio.sh`) by run.sh,
  # which is a read, not an exec; needs `r` for the same reason as
  # /init above.
  /usr/lib/bashio/** rix,

  # /data is the add-on's only writable persistent location (mapped by
  # `map: [data:rw]` in config.yaml). The DLQ JSONL files live here.
  /data/** rw,

  # Read-only access to system libraries / certificates needed by the
  # gluco-hub binary (glibc-linked, talks HTTPS via rustls).
  /etc/ssl/certs/** r,
  /etc/ca-certificates/** r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/hosts r,
  /usr/share/zoneinfo/** r,

  # Temporary files (used by config-crate parsing of options.json).
  /tmp/** rwk,

  # Devices used by the runtime.
  /dev/null rw,
  /dev/zero r,
  /dev/random r,
  /dev/urandom r,

  # Capabilities — only what gluco-hub actually needs.
  # No CAP_SYS_ADMIN, CAP_NET_ADMIN, CAP_DAC_*, no ptrace, no module loads.
  capability dac_override,
  capability net_bind_service,
  capability setgid,
  capability setuid,

  # Signals (s6 supervises gluco-hub).
  signal (send) set=(term,int,hup,cont,kill),
  signal (receive) set=(term,int,hup,cont,kill),

  # Network: outbound HTTPS to LibreLink Up + Nightscout, MQTT to the
  # internal Mosquitto add-on. No raw sockets, no SCTP.
  network inet stream,
  network inet6 stream,
  network unix stream,
  # DNS resolution.
  network inet dgram,
  network inet6 dgram,

  # Explicitly DENY common escape paths.
  deny /proc/sys/kernel/** w,
  deny /proc/kcore r,
  deny /sys/kernel/** w,
  deny /sys/firmware/** rwx,
  deny mount,
  deny umount,
  deny ptrace,
  deny capability sys_admin,
  deny capability sys_module,
  deny capability sys_rawio,
}
